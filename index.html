consulta.py 
import requests

# Configura tu API Key

RUC = "20601138572"

    "https://api.consultaperuapi.com/api/v1/consultas",

consulta_dni.py 
import requests

# Configura tu API Key
API_KEY = "tu_api_key"
DNI = "12345678"

response = requests.post(
    "https://api.consultaperuapi.com/api/v1/consultas-dni",

consulta.js 
// Configura tu API Key
const API_KEY = "tu_api_key";
const ruc = "20601138572";

const response = await fetch(
  "https://api.consultaperuapi.com/api/v1/consultas",
  {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ token: API_KEY, ruc })
  }
);

const data = await response.json();
console.log(data.razon_social);  // "EMPRESA EJEMPLO S.A.C."
console.log(data.estado);        // "ACTIVO"

consulta.php 
<?php

// Configura tu API Key
$apiKey = "tu_api_key";
$ruc    = "20601138572";
$ch = curl_init("https://api.consultaperuapi.com/api/v1/consultas");
curl_setopt_array($ch, [
    CURLOPT_RETURNTRANSFER => true,
    CURLOPT_POST           => true,
    CURLOPT_HTTPHEADER     => ["Content-Type: application/json"],
    CURLOPT_POSTFIELDS     => json_encode(["token" => $apiKey, "ruc" => $ruc])
]);

$body = curl_exec($ch);
$data = json_decode($body);

echo $data->razonSocial;  // "EMPRESA EJEMPLO S.A.C."
echo $data->estado;       // "ACTIVO"

<?php

// Configura tu API Key
$apiKey = "tu_api_key";
$dni    = "12345678";
$ch = curl_init("https://api.consultaperuapi.com/api/v1/consultas-dni");

    CURLOPT_RETURNTRANSFER => true,

Campos de la respuesta

ruc

Número de RUC del contribuyente (11 dígitos)

razonSocial

Nombre o razón social registrada en SUNAT

estado

Estado del contribuyente: ACTIVO, BAJA, etc.

condicion

Condición de domicilio: HABIDO, NO HABIDO, etc.

direccion

Domicilio fiscal registrado

ubigeo

Ubicación geográfica: Departamento, Provincia, Distrito

Campos de la respuesta

dni

Número de DNI consultado (8 dígitos)

nombres

Nombres de la persona

apellidoPaterno

Apellido paterno

apellidoMaterno

Apellido materno

nombreCompleto

Nombre completo formateado

direccion

Dirección del domicilio fiscal (puede ser null)

fuente

Fuente de datos: padron_local o callback
